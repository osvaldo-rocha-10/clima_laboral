public function EliminarRespuestaID($id)
{

      $sql = "DELETE FROM `respuestas` WHERE `respuestas`.`empleado` = :id";
      $query = $this->pdo->prepare($sql);
      $query->bindParam(":id", $id);
      $query->execute();
}/////////////////////////////////////////////////////////////////

}
